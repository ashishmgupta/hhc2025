#!/bin/bash
echo "Tropical Christmas Holiday Countdown"

current_day=$(date +"%j")
christmas_day=$(date -jf "%m-%d" "12-25" +"%j")

if [ "$christmas_day" -ge "$current_day" ]; then
  days_until_xmas=$((christmas_day - current_day))
  echo "Only $days_until_xmas days until Christmas on the island!"
else
  echo "Christmas has passed! Hope you enjoyed the sandy snowmen!"
fi
